def f(data):
    M = 1
    r = data[0]
    dr = data[1]
    r1=dr
    phi=data[2]
    p=phi
    dphi = data[3]
    p1=dphi
    
    ddr = ((4 * M**3) - (4 * M**2 * r) - (4 * M**2 * r**3 * p1**2) + (4 * M * r**4 * p1**2) - (r**5 * p1**2) + (r**2 * (M - 3 * M * r1**2))) / ((2 * M - r) * r**3)        
    ddphi = (2 * (-3 * M + r) * r1 * p1) / ((2 * M - r) * r)
    
    return np.array([dr, ddr, dphi, ddphi], float)

def trajectory(R):
    
    
    h = 0.01
    T = np.arange(0,1000, h)
    radius = []
    phi = []
    for t in T:
        radius.append(R[0])
        phi.append(R[2])
        R = R + h * f(R)
    xs = radius*np.cos(phi)
    ys = radius*np.sin(phi)
    return xs, ys
